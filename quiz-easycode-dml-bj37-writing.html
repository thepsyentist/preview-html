CREATE TABLE customers (
    id        INT PRIMARY KEY AUTO_INCREMENT,
    name      VARCHAR(50)  NOT NULL,
    email     VARCHAR(100) NOT NULL,
    age       INT,
    grade     VARCHAR(20) DEFAULT 'normal',
    joined_at DATE NOT NULL
);
